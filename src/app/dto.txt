// get timed elimination

enum TimedEliminationState {
    Round1 = 'Round1',
    Round2 = 'Round2',
    Finished = 'Finished'
}

// BE sorting by bibNumber, FE sorting by placement
interface GetTimedElimination {
    state: TimedEliminationState;
    isSecondRoundSkipped: boolean; // by default zwracasz false, chyba ze bedzie request
    participants: {
        id: string;
        bibNumber: number;
        name: string;
        firstRoundTime: number | null; // in miliseconds
        secondRoundTime: number | null; // in miliseconds
        placement: number | null
    }[];
}


// post timed elimination

// path params
interface PostTimedEliminationPathParams {
    eventId: string;
    categoryId: string;
}

// request body 
interface PostTimedEliminationRequestBody {
    state: TimedEliminationState
    participants: {
        id: string;
        roundTime: number | null; // in miliseconds
    }
}
